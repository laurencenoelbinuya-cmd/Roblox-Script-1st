local numberSuffixes = {
    [""] = 1,
    K = 1e3, M = 1e6, B = 1e9, T = 1e12,
    Qa = 1e15, Qi = 1e18, Sx = 1e21, Sp = 1e24,
    Oc = 1e27, No = 1e30, De = 1e33,
}

local function parseAbbreviatedNumber(input)
    if type(input) == "number" then
        return input
    end
    if type(input) ~= "string" then
        return 0
    end

    local numPart, suffix = input:match("^([%d%.]+)%s*(%a*)$")
    if not numPart then
        return tonumber(input) or 0
    end

    local multiplier = numberSuffixes[suffix] or 1
    return (tonumber(numPart) or 0) * multiplier
end

local function getCurrentWins()
    local leaderstats = player:FindFirstChild("leaderstats")
    if not leaderstats then return 0 end

    local stat = leaderstats:FindFirstChild("Wins")
        or leaderstats:FindFirstChild("Win")
        or leaderstats:FindFirstChild("Trophies")

    if stat then
        return parseAbbreviatedNumber(stat.Value)
    end
    return 0
end

-- Pet Deletion Variables
getgenv().autoDeleteCommon = false
getgenv().autoDeleteUncommon = false

getgenv().autoDeleteEpic = false
getgenv().autoDeleteLoopActive = false


local function handleAutoDelete()
    -- Prevent multiple loops from running at once
    if getgenv().autoDeleteLoopActive then return end

    -- If all toggles are off, don't start the loop
    if not (getgenv().autoDeleteCommon or getgenv().autoDeleteUncommon or getgenv().autoDeleteRare or getgenv().autoDeleteEpic) then return end

    getgenv().autoDeleteLoopActive = true
    task.spawn(function()
        while (getgenv().autoDeleteCommon or getgenv().autoDeleteUncommon or getgenv().autoDeleteRare or getgenv().autoDeleteEpic) do
            if GetPetsRemote and ChangePetRemote then
                -- Safely get pets (pcall prevents the script from breaking if the server errors)
                local success, myPets = pcall(function() return GetPetsRemote:InvokeServer() end)

                if success and type(myPets) == "table" then
                    local petsToDelete = {}

                    for _, petData in pairs(myPets) do
                        if petData.Equipped == false then
                            local rarity = petData.Rarity

                            if (rarity == "Common" and getgenv().autoDeleteCommon) or
                               (rarity == "Uncommon" and getgenv().autoDeleteUncommon) or
                               (rarity == "Rare" and getgenv().autoDeleteRare) or
                               (rarity == "Epic" and getgenv().autoDeleteEpic) then

                                table.insert(petsToDelete, petData.ID)
                            end
                        end
                    end

                    if #petsToDelete > 0 then
                        pcall(function() ChangePetRemote:InvokeServer(petsToDelete, "MultiDelete") end)
                    end
                end
            end
            task.wait(2) 
        end
        getgenv().autoDeleteLoopActive = false
    end)
end

FarmTab:CreateSection("Auto Farming")

FarmTab:CreateToggle({
   Name = "Auto Wall Break & Teleport",
   CurrentValue = false,
   Flag = "AutoBreakTP",
   Callback = function(Value)
       getgenv().autoBreakAndTeleport = Value
       if Value then
           task.spawn(function()
               local remote = game:GetService("ReplicatedStorage").Remotes.Events:FindFirstChild("WallBreakFinished")

               while getgenv().autoBreakAndTeleport do
                   local character = player.Character or player.CharacterAdded:Wait()
                   local rootPart = character:FindFirstChild("HumanoidRootPart")

                   if rootPart and remote then
                       -- 1. The Loop
                       for i = 1, 25 do
                           if not getgenv().autoBreakAndTeleport then break end
                           remote:FireServer()
                           task.wait(0.01)
                       end

                       -- 2. The Teleport
                       if getgenv().autoBreakAndTeleport then
                           local targetCFrame = worldCoordinates[getgenv().currentWorld] or worldCoordinates[1]
                           rootPart.CFrame = targetCFrame
                       end
                   end
                   task.wait(0.1)
               end
           end)
       end
   end,
})

FarmTab:CreateSection("Auto Rebirth")

FarmTab:CreateToggle({
   Name = "Auto Rebirth",
   CurrentValue = false,
   Flag = "AutoRebirth",
   Callback = function(Value)
       getgenv().autoRebirth = Value
       if Value then
           task.spawn(function()
               warn("[AutoRebirth] Loop started")
               while getgenv().autoRebirth do
                   local ok, wins = pcall(getCurrentWins)
                   if not ok then
                       warn("[AutoRebirth] Failed to read Wins stat:", wins)
                       task.wait(1)
                   elseif wins >= getgenv().requiredWinsForRebirth then
                       warn("[AutoRebirth] Wins =", wins, "-> firing RebirthRemote")
                       local fireOk, fireErr = pcall(function()
                           game:GetService("ReplicatedStorage").Remotes.Events.RebirthRemote:FireServer()
                       end)
                       if not fireOk then
                           warn("[AutoRebirth] FireServer failed:", fireErr)
                       end
                       task.wait(1)
                   else
                       warn("[AutoRebirth] Wins =", wins, "below threshold, waiting")
                       task.wait(0.5)
                   end
               end
               warn("[AutoRebirth] Loop stopped")
           end)
       end
   end,
})

PetsTab:CreateToggle({
   Name = "Auto Hatch (On/Off)",
   CurrentValue = false,
   Flag = "AutoHatchToggle",
   Callback = function(Value)
       getgenv().autoHatchEnabled = Value

       -- Suppress the pet reveal popup while auto hatching, restore it after.
       -- The game re-enables ShowCaseUI on its own each hatch, so we keep forcing it back off.
       local showCaseUI = player:WaitForChild("PlayerGui"):FindFirstChild("ShowCaseUI")
       if showCaseUI then
           if showCaseSuppressConn then
               showCaseSuppressConn:Disconnect()
               showCaseSuppressConn = nil
           end

           if Value then
               showCaseUI.Enabled = false
               showCaseSuppressConn = showCaseUI:GetPropertyChangedSignal("Enabled"):Connect(function()
                   if getgenv().autoHatchEnabled and showCaseUI.Enabled then
                       showCaseUI.Enabled = false
                   end
               end)
           else
               showCaseUI.Enabled = true
           end
       end

       -- The reveal animation also turns on a Lighting blur; keep it suppressed too
       local lighting = game:GetService("Lighting")
       if lightingBlurConn then
           lightingBlurConn:Disconnect()
           lightingBlurConn = nil
       end

       if Value then
           for _, effect in ipairs(lighting:GetChildren()) do
               if effect:IsA("BlurEffect") or effect:IsA("DepthOfFieldEffect") then
                   effect.Enabled = false
               end
           end
           lightingBlurConn = lighting.ChildAdded:Connect(function(child)
               if getgenv().autoHatchEnabled and (child:IsA("BlurEffect") or child:IsA("DepthOfFieldEffect")) then
                   child.Enabled = false
               end
           end)
       else
           for _, effect in ipairs(lighting:GetChildren()) do
               if effect:IsA("BlurEffect") or effect:IsA("DepthOfFieldEffect") then
                   effect.Enabled = true
               end
           end
       end

       if Value then
           task.spawn(function()
               while getgenv().autoHatchEnabled do
                   local args = { [1] = getgenv().selectedEggType, [2] = 1 }
                   game:GetService("ReplicatedStorage").Remotes.HatchPet:FireServer(unpack(args))
                   task.wait(0.1)
               end
           end)
       end
   end,
})
